#*****************************#
# Create Network if not exist #
#*****************************#

docker network create \
  --driver overlay \
  --subnet 192.168.100.0/24 \
  --gateway 192.168.100.1 \
  my-network


#****************************************#
# Run AUTH                               #
# Need assign uri, username and password #
#****************************************#

docker service create --name="auth-1" \
--env __PROP_spring_cloud_config_uri=http://config-server-1:8700 \
--env __PROP_spring_profiles_active=default \
--env __PROP_spring_cloud_config_label= \
--publish published=9011,target=8080 \
--with-registry-auth \
--network my-network \
docker-registry.eprocurement.systems/e-auth:17.12.1
